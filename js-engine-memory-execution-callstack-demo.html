const x = 2;
const getSum = (n) => {
  let sum = 0;
  for (let i = 1; i <= n; i++) {
    sum += i;
  }
  return sum;
};

console.log(getSum(x)); // Output: 3
console.log(getSum(5)); // Output: 15
